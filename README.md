# install/start

$ sudo apt update && apt upgrade -y <br>

$ sudo apt install git <br>

$ git clone https://github.com/paraparap/getgeoinfo <br>

$ cd getgeoinfo <br>

$ sudo chmod +x install.sh <br>

$ sudo ./install.sh <br>

$ sudo ./geoinfo.py

# and in new terminal

$ cd getgeoinfo <br>

$ sudo ./ngrok http 8080
